```python
@app.function(image=train_image, volumes={PROJECT: project_volume}, gpu="T4", timeout=3600)
def tokenize_dinov2_things(split: str = "train") -> None:
    ensure_fourm()
    import sys; sys.path.insert(0, f"{REPO}/ml-4m")
    import torch, tarfile, io, numpy as np
    from pathlib import Path
    from fourm.vq.vqvae import VQVAE
    import torchvision.transforms as T

    device = torch.device("cuda")

    # Load DINOv2-B/14 from torch hub
    dino = torch.hub.load("facebookresearch/dinov2", "dinov2_vitb14").to(device).eval()

    # Load 4M's DINOv2 VQVAE tokenizer
    vqvae = VQVAE.from_pretrained("EPFL-VILAB/4M_tokenizers_DINOv2-B14_8k_224-448").to(device).eval()

    preprocess = T.Compose([
        T.Resize(224, interpolation=T.InterpolationMode.BICUBIC),
        T.CenterCrop(224),
        T.ToTensor(),
        T.Normalize(mean=(0.485, 0.456, 0.406), std=(0.229, 0.224, 0.225)),
    ])

    images_root = Path(PROJECT) / "data" / split / "things" / "rgb"
    out_root    = Path(PROJECT) / "data" / split / "things" / "tok_dinov2@224"
    out_root.mkdir(parents=True, exist_ok=True)

    for shard_path in sorted(images_root.glob("shard_*.tar")):
        shard_idx = shard_path.stem.split("_")[1]
        out_path  = out_root / f"shard_{shard_idx}.tar"
        if out_path.exists():
            continue

        samples = {}  # key → (B, 256) int16
        with tarfile.open(shard_path) as tf:
            for member in tf.getmembers():
                if not member.name.endswith(".jpg"):
                    continue
                key = member.name.split(".")[0]
                buf = tf.extractfile(member).read()
                from PIL import Image
                img = Image.open(io.BytesIO(buf)).convert("RGB")
                samples[key] = preprocess(img).unsqueeze(0)  # (1, 3, 224, 224)

        import wds; writer = wds.TarWriter(str(out_path))
        for key, img_t in samples.items():
            img_t = img_t.to(device)
            with torch.no_grad():
                feats = dino.forward_features(img_t)
                patch_tokens = feats["x_norm_patchtokens"]       # (1, 256, 768)
                patch_tokens = patch_tokens.reshape(1, 16, 16, 768).permute(0, 3, 1, 2)  # (1, 768, 16, 16)
                tok = vqvae.tokenize(patch_tokens)               # (1, 16, 16)
                tok = tok.reshape(1, 256).cpu().numpy().astype(np.int16)  # (1, 256)
            buf = io.BytesIO(); np.save(buf, tok); buf.seek(0)
            writer.write({"__key__": key, "npy": buf.read()})
        writer.close()
        print(f"Wrote {out_path}", flush=True)

    project_volume.commit()
    print(f"Done tokenizing {split} split.", flush=True)
```

    for shard_path in shard_paths:
