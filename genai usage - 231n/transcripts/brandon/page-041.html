#!/usr/bin/env python3
"""Average a directory of raw .npy depth maps and save a plasma-coloured PNG.

Supports two layouts:
  - WDS tar shards:  shard_NNN.tar files, each containing {key}.npy entries
  - Flat directory:  plain .npy files (optionally recursive)

Each array should be 2-D (H, W) float or integer.  Arrays are min-max
normalised to [0, 1] before averaging so that differences in absolute depth
scale don't skew the result.

Usage:
    # WDS tar shards (default)
    python average_depth_maps.py --depth_dir /path/to/tok_depth

    # Flat .npy files
    python average_depth_maps.py --depth_dir /path/to/depth_maps --flat

    # Recurse into subdirectories (flat mode only)
    python average_depth_maps.py --depth_dir /path/to/depth_maps --flat --recursive

    # Also save the raw float average as a .npy file
    python average_depth_maps.py --depth_dir /path/to/tok_depth --save_npy
"""

from __future__ import annotations

import argparse
import io
import tarfile
from pathlib import Path

import matplotlib
matplotlib.use("Agg")
import matplotlib.pyplot as plt
import numpy as np


def _accumulate(running_sum, count, arr, normalize):
    arr = arr.astype(np.float32)
    if arr.ndim == 3:
        arr = arr.squeeze()
    if arr.ndim != 2:
        return running_sum, count, False
    if normalize:
        lo, hi = arr.min(), arr.max()
        arr = (arr - lo) / (hi - lo) if hi > lo else np.zeros_like(arr)
    if running_sum is None:
        running_sum = arr.astype(np.float64)
    else:
        if running_sum.shape != arr.shape:
            arr = _resize(arr, running_sum.shape)
        running_sum += arr.astype(np.float64)
    return running_sum, count + 1, True


def iter_arrays_from_shards(depth_dir: Path) -> int:
    """Yield (key, array) from all shard_NNN.tar files in depth_dir."""
    tar_paths = sorted(depth_dir.glob("shard_*.tar"))
    if not tar_paths:
        raise FileNotFoundError(f"No shard_*.tar files found in {depth_dir}")
    print(f"Found {len(tar_paths)} shards.", flush=True)
    for tar_path in tar_paths:
        with tarfile.open(tar_path) as tf:
            for member in tf.getmembers():
                if not member.name.endswith(".npy"):
                    continue
                buf = tf.extractfile(member).read()
                yield member.name, np.load(io.BytesIO(buf))


def iter_arrays_from_flat(depth_dir: Path, recursive: bool):
    """Yield (filename, array) from plain .npy files."""
    pattern = "**/*.npy" if recursive else "*.npy"
    paths = sorted(depth_dir.glob(pattern))
    if not paths:
        raise FileNotFoundError(f"No .npy files found in {depth_dir}")
    print(f"Found {len(paths)} .npy files.", flush=True)
    for p in paths:
        yield p.name, np.load(p)


def average_depth_maps(
    depth_dir: Path,
    flat: bool = False,
    recursive: bool = False,
    normalize: bool = True,
) -> tuple[np.ndarray, int]:
    """Load and average depth maps from shards or flat files.

    Returns (average_image, n_loaded) where average_image is float32 in [0, 1].
    """
    running_sum: np.ndarray | None = None
    count = 0

    source = iter_arrays_from_flat(depth_dir, recursive) if flat else iter_arrays_from_shards(depth_dir)

    for name, arr in source:
        running_sum, count, ok = _accumulate(running_sum, count, arr, normalize)
        if not ok:
            print(f"  Skipping {name}: unexpected shape {arr.shape}")
        if count % 500 == 0 and count > 0:
            print(f"  {count} loaded…", flush=True)

    if running_sum is None or count == 0:
        raise RuntimeError("No valid depth maps were loaded.")

    return (running_sum / count).astype(np.float32), count
