def assemble_figure_cpu(
    output_dir: str,
    model_names: list[str],
    sample_keys: list[str] | None = None,
    n_keys: int = 2,
) -> None:
    """Assemble a single comparison figure across all models.

    Layout:
      columns = selected sample keys  (default: first n_keys=2 from what was generated)
      rows    = reference rows (rgb, depth_gt) followed by one depth_pred row per model

    Depth images (grayscale L PNGs) are rendered with the plasma colormap.
    Output: {output_dir}/comparison_figure.png
    """
    import json
    from pathlib import Path

    import matplotlib
    matplotlib.use("Agg")
    import matplotlib.pyplot as plt
    import numpy as np
    from PIL import Image as PILImage

    output_dir_path = Path(output_dir)

    # ── Load per-model metadata ───────────────────────────────────────────
    metas = {}
    for model_name in model_names:
        meta_path = output_dir_path / model_name / "meta.json"
        if not meta_path.exists():
            print(f"  Skipping {model_name}: meta.json not found", flush=True)
            continue
        metas[model_name] = json.loads(meta_path.read_text())

    if not metas:
        print("No valid model directories found — nothing to plot.", flush=True)
        return

    # Keys to display: explicit list, or first n_keys from the first model
    first_meta = next(iter(metas.values()))
    display_keys = sample_keys or first_meta["keys"][:n_keys]
    print(f"  Displaying keys: {display_keys}", flush=True)

    # Reference images (rgb, depth_gt) are identical across models — use first model
    ref_dir = output_dir_path / next(iter(metas))
    ref_rows = ["rgb", "depth_gt"]
    # Only include dino_gt reference row if any model generated it
    if any("dino_gt" in m["image_types"] for m in metas.values()):
        ref_rows = ["rgb", "dino_gt", "depth_gt"]

    model_row_names = list(metas.keys())
    row_labels = ref_rows + model_row_names
    n_rows = len(row_labels)
    n_cols = len(display_keys)

    fig, axes = plt.subplots(
        n_rows, n_cols,
        figsize=(3.5 * n_cols, 3.0 * n_rows),
        squeeze=False,
    )

    def _show(ax, img_path: Path, itype: str) -> None:
        if not img_path.exists():
            ax.text(0.5, 0.5, "missing", ha="center", va="center", transform=ax.transAxes)
            ax.axis("off")
            return
        arr = np.array(PILImage.open(img_path))
        if "depth" in itype:
            ax.imshow(arr, cmap="plasma", vmin=0, vmax=255)
        else:
            ax.imshow(arr)
        ax.axis("off")

    # ── Reference rows ────────────────────────────────────────────────────
    for row, itype in enumerate(ref_rows):
        for col, key in enumerate(display_keys):
            ax = axes[row][col]
            _show(ax, ref_dir / f"{key}_{itype}.png", itype)
            if col == 0:
                ax.set_ylabel(itype.replace("_", " "), fontsize=9,
                              rotation=0, labelpad=55, va="center")
        axes[row][0].yaxis.label.set_fontweight("bold")

    # ── Column headers (key labels on top row) ────────────────────────────
    for col, key in enumerate(display_keys):
        axes[0][col].set_title(f"key {key}", fontsize=9)

    # ── One depth_pred row per model ──────────────────────────────────────
    for m_idx, (model_name, meta) in enumerate(metas.items()):
        row = len(ref_rows) + m_idx
        for col, key in enumerate(display_keys):
            _show(axes[row][col], output_dir_path / model_name / f"{key}_depth_pred.png", "depth")
        cond = meta.get("conditions", {})
        cond_str = " ".join(f"+{k}" for k, v in cond.items() if v)
        label = model_name + (f"\n{cond_str}" if cond_str else "")
        axes[row][0].set_ylabel(label, fontsize=9, rotation=0, labelpad=55, va="center")

    plt.tight_layout()
    out = output_dir_path / "comparison_figure.png"
    plt.savefig(out, dpi=150, bbox_inches="tight")
    plt.close(fig)
    print(f"  Saved → {out}", flush=True)

    project_volume.commit()
    print("Figure assembly complete.")

#!/usr/bin/env python3
"""Average a directory of raw .npy depth maps and save a plasma-coloured PNG.

Each .npy file should contain a 2-D (H, W) float or integer depth array.
Arrays are min-max normalised to [0, 1] before averaging so that differences
in absolute depth scale don't skew the result.

Usage:
    python average_depth_maps.py --depth_dir /path/to/depth_maps

    # Recurse into subdirectories
    python average_depth_maps.py --depth_dir /path/to/depth_maps --recursive

    # Also save the raw float average as a .npy file
    python average_depth_maps.py --depth_dir /path/to/depth_maps --save_npy

    # Skip per-image normalisation (use raw values)
    python average_depth_maps.py --depth_dir /path/to/depth_maps --no_normalize
"""

from __future__ import annotations

import argparse
from pathlib import Path

import matplotlib
matplotlib.use("Agg")
import matplotlib.pyplot as plt
import numpy as np


def collect_npy_paths(depth_dir: Path, recursive: bool) -> list[Path]:
    pattern = "**/*.npy" if recursive else "*.npy"
    paths = sorted(depth_dir.glob(pattern))
    if not paths:
        raise FileNotFoundError(f"No .npy files found in {depth_dir} (recursive={recursive})")
    return paths


def average_depth_maps(
    paths: list[Path],
    normalize: bool = True,
) -> tuple[np.ndarray, int]:
    """Load and average depth maps.

    Returns (average_image, n_loaded) where average_image is float32 [0, 1].
    """
    running_sum: np.ndarray | None = None
    count = 0

    for p in paths:
        arr = np.load(p).astype(np.float32)
        if arr.ndim == 3:
            arr = arr.squeeze()   # drop a leading batch/channel dim if present
        if arr.ndim != 2:
            print(f"  Skipping {p.name}: unexpected shape {arr.shape}")
            continue

        if normalize:
            lo, hi = arr.min(), arr.max()
            if hi > lo:
                arr = (arr - lo) / (hi - lo)
            else:
                arr = np.zeros_like(arr)

        if running_sum is None:
            running_sum = arr.astype(np.float64)
        else:
            if running_sum.shape != arr.shape:
                arr = _resize(arr, running_sum.shape)
            running_sum += arr.astype(np.float64)
        count += 1

        if count % 500 == 0:
            print(f"  {count}/{len(paths)} loaded…", flush=True)

    if running_sum is None or count == 0:
        raise RuntimeError("No valid depth maps were loaded.")

    avg = (running_sum / count).astype(np.float32)
    return avg, count


def _resize(arr: np.ndarray, target_shape: tuple[int, int]) -> np.ndarray:
    """Resize a 2-D array to target_shape using PIL (bilinear)."""
    from PIL import Image
    h, w = target_shape
    pil = Image.fromarray(arr).resize((w, h), Image.BILINEAR)
    return np.array(pil)


def save_plot(avg: np.ndarray, n: int, out_path: Path) -> None:
    fig, ax = plt.subplots(1, 1, figsize=(5, 5))
    ax.imshow(avg, cmap="plasma", vmin=0, vmax=1)
    ax.set_title(f"Average GT depth  (n={n:,})", fontsize=12)
    ax.axis("off")
    plt.tight_layout()
    out_path.parent.mkdir(parents=True, exist_ok=True)
    plt.savefig(out_path, dpi=150, bbox_inches="tight")
    plt.close(fig)
    print(f"Saved plot → {out_path}")

    save_plot(avg, n, out_png)

    if args.save_npy:
        npy_out = out_png.with_suffix(".npy")
        np.save(npy_out, avg)
        print(f"Saved array → {npy_out}")


if __name__ == "__main__":
    main()
