288	def decode_to_image(
289	    tokenizer: DiVAE,
290	    tokens: torch.Tensor,
291	    is_rgb: bool,
292	    timesteps: int = 25,
293	    image_size: int = 224,
294	) -> np.ndarray:
295	    """Decode integer tokens (1, 196) → numpy image array for display.
296	
297	    RGB: returns (224, 224, 3) float32 in [0, 1].
298	    Depth: returns (224, 224) float32 in [0, 1], min-max normalised.
299	    timesteps: diffusion steps for DiVAE decoder (default 25, matching 4M demo;
300	               1000 train-time default hangs for minutes).
301	    """
302	    tokens_2d = tokens.reshape(1, 14, 14)
303	    with torch.no_grad():
304	        img = tokenizer.decode_tokens(tokens_2d, timesteps=timesteps, image_size=image_size)
305	    img = img.squeeze(0).cpu().float()
306	    if is_rgb:
307	        img = img * 0.5 + 0.5                      # [-1, 1] → [0, 1]
308	        img = img.permute(1, 2, 0).numpy()         # (224, 224, 3)
309	        return np.clip(img, 0.0, 1.0)
310	    else:
311	        img = img.squeeze(0).numpy()               # (224, 224)
312	        lo, hi = img.min(), img.max()
313	        if hi > lo:
314	            img = (img - lo) / (hi - lo)           # min-max → [0, 1]
315	        return img.clip(0.0, 1.0)

318	def decode_dinov2(
319	    tokenizer,           # VQVAE (not DiVAE — no diffusion, fast)
320	    tokens: torch.Tensor,  # (1, 256) int64
321	    image_size: int = 224,
322	) -> np.ndarray:
323	    """Decode DINOv2 tokens → (224, 224, 3) float32 PCA feature visualization.
324	
325	    Decodes to (1, 768, 16, 16) feature map, applies 3-component PCA across
326	    the spatial dimension, normalises to [0, 1], then upsamples to image_size.
327	    """
328	    from sklearn.decomposition import PCA
329	    from PIL import Image as _Image
330	
331	    tokens_2d = tokens.reshape(1, 16, 16)
332	    with torch.no_grad():
333	        feats = tokenizer.decode_tokens(tokens_2d)   # (1, 768, 16, 16)
334	    feats = feats.squeeze(0).cpu().float().numpy()   # (768, 16, 16)
335	    _, H, W = feats.shape
336	    feats_flat = feats.reshape(feats.shape[0], -1).T  # (256, 768)
337	
338	    pca = PCA(n_components=3)
339	    pca_out = pca.fit_transform(feats_flat).reshape(H, W, 3)  # (16, 16, 3)
340	    lo, hi = pca_out.min(), pca_out.max()
341	    if hi > lo:
342	        pca_out = (pca_out - lo) / (hi - lo)
343	
344	    pca_uint8 = (pca_out.clip(0, 1) * 255).astype(np.uint8)
345	    img = _Image.fromarray(pca_uint8).resize((image_size, image_size), _Image.NEAREST)
346	    return np.array(img).astype(np.float32) / 255.0

def assemble_figure_cpu(
    output_dir: str,
    model_names: list[str],
    sample_keys: list[str] | None = None,
) -> None:
    """Read saved individual PNGs and assemble a per-model grid figure.

    Rows = samples, columns = image types (rgb, depth_gt, depth_pred, dino_gt).
    Colormaps are applied at plot time: depth images are saved as grayscale L and
    rendered with 'plasma'; color images are displayed as-is.

    Figures are saved to {output_dir}/{model_name}_figure.png.
    """
    import json
    from pathlib import Path

    import matplotlib
    matplotlib.use("Agg")
    import matplotlib.pyplot as plt
    import numpy as np
    from PIL import Image as PILImage

    output_dir_path = Path(output_dir)

    for model_name in model_names:
        model_dir = output_dir_path / model_name
        meta_path = model_dir / "meta.json"
        if not meta_path.exists():
            print(f"  Skipping {model_name}: meta.json not found at {meta_path}", flush=True)
            continue

        meta = json.loads(meta_path.read_text())
        keys        = sample_keys or meta["keys"]
        image_types = meta["image_types"]
        colormaps   = meta["colormaps"]

        n_rows = len(keys)
        n_cols = len(image_types)
        fig, axes = plt.subplots(n_rows, n_cols, figsize=(3.5 * n_cols, 3.5 * n_rows), squeeze=False)

        col_titles = [t.replace("_", " ") for t in image_types]

        for row, key in enumerate(keys):
            for col, itype in enumerate(image_types):
                img_path = model_dir / f"{key}_{itype}.png"
                ax = axes[row][col]
                if not img_path.exists():
                    ax.text(0.5, 0.5, "missing", ha="center", va="center", transform=ax.transAxes)
                    ax.axis("off")
                    continue

                img = PILImage.open(img_path)
                arr = np.array(img)

                if colormaps.get(itype) == "plasma":
                    # Grayscale L image — apply colormap
                    ax.imshow(arr, cmap="plasma", vmin=0, vmax=255)
                else:
                    ax.imshow(arr)

                if row == 0:
                    ax.set_title(col_titles[col], fontsize=10)
                ax.axis("off")

            axes[row][0].set_ylabel(
                f"key {key}", fontsize=8, rotation=0, labelpad=50, va="center"
            )

        cond = meta.get("conditions", {})
        cond_str = " | ".join(f"{k}={v}" for k, v in cond.items() if v)
        title = model_name + (f"\n({cond_str})" if cond_str else "")
        fig.suptitle(title, fontsize=12, fontweight="bold")
        plt.tight_layout()

        out = output_dir_path / f"{model_name}_figure.png"
        plt.savefig(out, dpi=150, bbox_inches="tight")
        plt.close(fig)
        print(f"  Saved figure → {out}", flush=True)

    project_volume.commit()
    print("Figure assembly complete.")
