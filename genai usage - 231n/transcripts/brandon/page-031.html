- 3 intermediate evals (indices ≈11, 22, 32) give ~1.7B / 3.4B / 5B token checkpoints, all above the 1B filter → **12 (N,D,loss) data points** per condition for Chinchilla fitting. Good.

### Budget Estimate
- Single A100 @ ~$3.50/hr
- Rough step time: ~10ms (7M model) to ~40ms (44M model)  
- Steps per run: 5e9 / (64×512) ≈ 152,600 steps
- Wall time: ~0.4 hr (tiny) to ~1.7 hr (largest) → ~4.5 hrs per condition × 2 conditions = **~9 GPU hours ≈ $32**
- Well within $175. Budget is comfortable; even doubling estimates leaves headroom.

### Recommendations
- **GPU**: Keep A100. It's the right call—smaller GPUs (L4/A10G) save ~30% cost but risk OOM at larger batch sizes.
- **Batch size**: Consider increasing from 64 → 128 to halve GPU hours. The `blr` already scales linearly with batch size in `run_training_4m.py` (`lr = blr * batch_size / 256`), so `blr: 0.0001` stays correct for either.
- **Run order**: Run the largest model first to validate the full pipeline before committing all budget.
- **`save_ckpt_freq`**: Already set to 2, meaning checkpoints every 2 epochs. Fine for result collection.
- **Fitting**: Chinchilla needs ≥3 points; 12 points is healthy. The mixed-depth design (2e2d vs 3e3d) adds noise to the alpha/beta fit but won't prevent a visible trend.

def _read_modality_tar(path: Path) -> dict[str, np.ndarray]:
    """Return {key: array} for every .npy file in a single-modality THINGS shard."""
    samples: dict[str, np.ndarray] = {}
    with tarfile.open(path) as tf:
        for member in tf.getmembers():
            if not member.name.endswith(".npy"):
                continue
            # Key is everything before the first dot: "000042" from "000042.tok_rgb.npy"
            key = member.name.split(".")[0]
            buf = tf.extractfile(member).read()
            samples[key] = np.load(io.BytesIO(buf))
    return samples


def load_things_samples(
    things_root: Path,
    shard_idx: int,
    n_samples: int,
    seed: int = 42,
) -> list[dict[str, Any]]:
    """Load tok_rgb and tok_depth numpy arrays from THINGS shard tars.

    THINGS shards are stored one modality per tar:
      things_root/tok_rgb/shard_NNN.tar
      things_root/tok_depth/shard_NNN.tar
    Each contains flat (196,) int16 arrays (no augmentation axis).
    """
    rgb_tar   = things_root / "tok_rgb"   / f"shard_{shard_idx:03d}.tar"
    depth_tar = things_root / "tok_depth" / f"shard_{shard_idx:03d}.tar"

    for p in (rgb_tar, depth_tar):
        if not p.exists():
            raise FileNotFoundError(f"Shard not found: {p}")

    rgb_samples   = _read_modality_tar(rgb_tar)
    depth_samples = _read_modality_tar(depth_tar)

    common_keys = sorted(set(rgb_samples) & set(depth_samples))
    if not common_keys:
        raise RuntimeError("No common keys between tok_rgb and tok_depth shards.")

    rng = random.Random(seed)
    selected = rng.sample(common_keys, min(n_samples, len(common_keys)))

    return [
        {
            "key": k,
            "tok_rgb":   rgb_samples[k].astype(np.int64),    # (196,) int64
            "tok_depth": depth_samples[k].astype(np.int64),  # (196,) int64
        }
        for k in selected
    ]

def decode_to_image(
    tokenizer: DiVAE,
    tokens: torch.Tensor,
    is_rgb: bool,
) -> np.ndarray:
    """Decode integer tokens (1, 196) → numpy image array for display.

    RGB: returns (224, 224, 3) float32 in [0, 1].
    Depth: returns (224, 224) float32 in [0, 1].
    """
    tokens_2d = tokens.reshape(1, 14, 14)
    with torch.no_grad():
        img = tokenizer.decode_tokens(tokens_2d)   # (1, C, 224, 224)
    img = img.squeeze(0).cpu().float()
    if is_rgb:
        img = img * 0.5 + 0.5                      # [-1, 1] → [0, 1]
        img = img.permute(1, 2, 0).numpy()         # (224, 224, 3)
    else:
        img = img.squeeze(0).numpy()               # (224, 224)
    return np.clip(img, 0.0, 1.0)
