"""Run the named validation-task suite on the *live* model during training.

Stock 4M's in-loop eval can only validate the training task (its val loaders
derive masking from the train dataset of the same name). To also score the
named tasks in ``configs/4m_things_val_tasks.yaml`` every ``eval_freq`` epochs,
we wrap two module-global functions of the stock trainer:

  * ``train_one_epoch`` → records the current epoch (so we can dedup).
  * ``evaluate``        → after stock eval, runs the suite *once per eval epoch*
                          and merges per-task loss into the returned stats, so
                          they flow into the trainer's ``log.txt`` / wandb.

The wrapper *factories* below are pure (no fourm) and unit-tested. The fourm-
dependent suite builder (``build_suite_fn``) reuses ``validate_4m`` wholesale —
it does not re-implement validation.

This module is installed from ``train_4m.run_train`` by importing the stock
trainer (not running it as ``__main__``) and reassigning the two globals before
``main(args)`` is called — see that function for the launch wiring.
"""

from __future__ import annotations

from pathlib import Path
from typing import Any, Callable

_HERE = Path(__file__).resolve().parent
_REPO_ROOT = _HERE.parent.parent


class _EpochCell:
    """Shared mutable holder for the current training epoch."""

    def __init__(self) -> None:
        self.epoch: int = -1


def make_epoch_tracking_train(orig_train_one_epoch: Callable, cell: _EpochCell) -> Callable:
    """Wrap ``train_one_epoch`` so it records the epoch it was called with.

    Stock calls it as ``train_one_epoch(..., epoch=epoch, ...)`` once per epoch.
    """

    def wrapped(*args: Any, **kwargs: Any):
        if "epoch" in kwargs:
            cell.epoch = kwargs["epoch"]
        return orig_train_one_epoch(*args, **kwargs)

    return wrapped


def make_suite_evaluate(
    orig_evaluate: Callable,
    suite_fn: Callable[[Any, Any], dict],
    cell: _EpochCell,
    state: dict,
) -> Callable:
    """Wrap ``evaluate`` to also run the task suite once per eval epoch.

    Stock calls ``evaluate(model, data_loader, device, ...)`` up to twice per
    eval epoch (val loader, then fixed-eval loader). We run ``suite_fn`` only on
    the first call for a given epoch (tracked via ``cell``), merge its stats into
    the stock result, and swallow suite errors so a val hiccup never kills a run.
    """

    def wrapped(model, data_loader, device, *args: Any, **kwargs: Any):
        stats = orig_evaluate(model, data_loader, device, *args, **kwargs)
        if cell.epoch != state["last_epoch"]:
            state["last_epoch"] = cell.epoch
            try:
                stats = {**stats, **suite_fn(model, device)}
            except Exception as exc:  # noqa: BLE001 — never crash training on val
                print(f"[in-loop val] suite failed at epoch {cell.epoch}: {exc!r}")
        return stats

    return wrapped


def install_in_loop_validation(trainer, suite_fn: Callable[[Any, Any], dict]) -> None:
    """Reassign the trainer's ``train_one_epoch`` / ``evaluate`` globals to the
    wrapped versions. ``main()`` looks both up as module globals at call time, so
    the patched functions take effect."""
    cell = _EpochCell()
    state = {"last_epoch": -1}
    trainer.train_one_epoch = make_epoch_tracking_train(trainer.train_one_epoch, cell)
    trainer.evaluate = make_suite_evaluate(trainer.evaluate, suite_fn, cell, state)

class ValidationSuite:
    """Run the named validation tasks on a model → ``{task_name: stats}``.

    The single home for "run the task suite on a model". It is reusable across
    calls: deterministic WDS loaders are built lazily and cached, and the set of
    runnable tasks is resolved once against the model's actual domains. Two
    callers share it, differing only in *where the model comes from* and *how the
    result is formatted*:

      * ``run_validation`` (CLI) — builds a model + loads a checkpoint, runs once.
      * ``in_loop_val`` (training) — runs on the live model every eval epoch.
    """

    def __init__(self, tasks_cfg: dict, *, input_size: int, text_tokenizer,
                 loss_type: str, batch_size: int, n_batches: int,
                 select: list[str] | None = None):
        self.tasks_cfg = tasks_cfg
        self.tasks = tasks_cfg["tasks"]
        names = select or list(self.tasks)
        unknown = [n for n in names if n not in self.tasks]
        if unknown:
            raise SystemExit(f"unknown task(s) {unknown}; available: {list(self.tasks)}")
        self.names = names
        self.input_size = input_size
        self.text_tokenizer = text_tokenizer
        self.loss_type = loss_type
        self.batch_size = batch_size
        self.n_batches = n_batches
        self.n_in = tasks_cfg.get("fixed_eval_input_tokens", 128)
        self.n_out = tasks_cfg.get("fixed_eval_target_tokens", 128)
        self._cache: dict[str, tuple] = {}
        self._runnable: list[str] | None = None

    def _loader_for(self, name: str):
        if name not in self._cache:
            task_cfg = _task_dataset_config(self.tasks[name], self.tasks_cfg)
            self._cache[name] = _build_task_loader(
                task_cfg, self.input_size, (self.n_in, self.n_in), (self.n_out, self.n_out),
                self.text_tokenizer, num_workers=0, batch_size=self.batch_size,
            )
        return self._cache[name]

    def _resolve_runnable(self, model) -> list[str]:
        """Keep only tasks the model can run: encoder needs every in_domain, decoder
        every out_domain. Reads the (possibly DDP-wrapped) model's embedding keys."""
        m = getattr(model, "module", model)
        enc = set(getattr(m, "encoder_embeddings", {}).keys())
        dec = set(getattr(m, "decoder_embeddings", {}).keys())
        runnable = []
        for name in self.names:
            task = self.tasks[name]
            missing = (set(task["in_domains"].split("-")) - enc) | (
                set(task["out_domains"].split("-")) - dec)
            if missing:
                print(f"[val suite] skipping '{name}': model lacks {sorted(missing)}")
            else:
                runnable.append(name)
        return runnable

    def run(self, model, device, dtype) -> dict[str, dict[str, Any]]:
        if self._runnable is None:
            self._runnable = self._resolve_runnable(model)
        results: dict[str, dict[str, Any]] = {}
        for name in self._runnable:
            loader, all_d = self._loader_for(name)
            results[name] = evaluate_task(
                model, loader, device, all_d, self.n_in, self.n_out,
                self.loss_type, dtype, self.n_batches,
            )
        return results
