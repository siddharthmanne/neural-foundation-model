1	"""§5.5 model-free retrieval probe.
2	
3	Does the token *representation* place trials of the same class nearer to each
4	other than to trials of other classes? This is asked **without** training a
5	classifier, so it cleanly separates "token quality is bad" from "the trained
6	probe (linear / MLP) was the bottleneck."
7	
8	For each query trial we compute the K nearest neighbors in feature space and
9	measure **precision@K**: the fraction of retrieved neighbors that share the
10	query's class label. We compute it three ways for parity with the §5.3 probe:
11	
12	  - tokens : same featurization the §5.3 probe uses on this tokenizer
13	             (mean of `tokens_to_embedding` if available; bag-of-codes else).
14	             For BrainOmni: dense embeddings → cosine similarity.
15	             For μ-transform with the new `tokens_to_embedding`: bin-center
16	             values → cosine similarity in (channel × time) feature space.
17	  - raw    : flattened signal x — same upper-bound bracket as §5.3.
18	  - random : uniformly random tokens, same shape — same lower bracket.
19	
20	Also reported for BrainOmni-style sparse codecs:
21	  - tokens_jaccard : Jaccard similarity on token *sets* (vocab-subset used per
22	                     trial). Position-blind by design, but matches the user's
23	                     mental model of "do similar trials share codewords?"
24	
25	Chance precision = class_frequency averaged over queries, exactly. With class
26	imbalance we report **balanced precision@K**: average of per-class precision
27	contributions weighted equally per class.
28	
29	Why this is the right escape valve from the §5.3 result:
30	  If `tokens` retrieval precision is at chance, no trained classifier (linear,
31	  MLP, transformer) can pull signal out of features that don't have it — the
32	  feature geometry simply doesn't separate classes. If retrieval precision is
33	  well above chance but §5.3 sits at chance, the bottleneck is the §5.3
34	  classifier (architecture / capacity / regularization), not the tokens.

170	def _jaccard_makes_sense(token_sets: torch.Tensor, codebook_size: int) -> bool:
171	    """Heuristic: skip Jaccard when token sets are saturated.
172	
173	    If the median trial uses >80% of the vocab, Jaccard ≈ 1 for every pair
174	    and the metric collapses to chance regardless of token quality. This is
175	    the μ-transform regime (dense per-(channel, time) codecs) where the
176	    cosine retrieval on the *value* featurization is the right diagnostic.
177	    """
178	    used_per_trial = token_sets.to(torch.int64).sum(dim=1).to(torch.float64)
179	    median_frac = float(used_per_trial.median().item()) / max(1, codebook_size)
180	    return median_frac <= 0.80

183	@torch.no_grad()
184	def _jaccard_precision_at_k(
185	    token_sets: torch.Tensor,  # (B, V) bool
186	    labels: torch.Tensor,
187	    Ks: tuple[int, ...],
188	    n_classes: int,
189	) -> dict[int, float]:
190	    """Pairwise Jaccard similarity, k-NN retrieval, balanced precision@K."""
191	    A = token_sets.to(torch.float32)
192	    # |A ∩ B| = A @ B^T (bool counts)
193	    intersection = A @ A.T
194	    set_size = A.sum(dim=1, keepdim=True)
195	    # |A ∪ B| = |A| + |B| - |A ∩ B|
196	    union = set_size + set_size.T - intersection
197	    jaccard = torch.where(union > 0, intersection / union, torch.zeros_like(union))
198	    return _precision_at_k_from_similarity(jaccard, labels, Ks, n_classes)

206	    feats: torch.Tensor,  # (B, D) float
207	    labels: torch.Tensor,
208	    Ks: tuple[int, ...],
209	    n_classes: int,
210	) -> dict[int, float]:
211	    """L2-normalize features, similarity = X @ X^T, balanced precision@K."""
212	    norms = feats.norm(dim=1, keepdim=True).clamp(min=1e-8)
213	    Xn = feats / norms
214	    sim = Xn @ Xn.T
215	    return _precision_at_k_from_similarity(sim, labels, Ks, n_classes)

221	def _precision_at_k_from_similarity(
222	    sim: torch.Tensor,  # (B, B) float
223	    labels: torch.Tensor,
224	    Ks: tuple[int, ...],
225	    n_classes: int,
226	) -> dict[int, float]:
227	    """For each row (query), top-K excluding self, fraction matching label.
228	    Reported as **balanced** precision: mean over classes of per-class avg.
229	
230	    Balanced because class imbalance otherwise lets a method that "retrieves
231	    similar things" score high on the majority class and drag the average up,
232	    even if minority classes are unrecoverable.
233	    """
234	    B = sim.shape[0]
235	    self_mask = torch.eye(B, dtype=torch.bool)
236	    sim_masked = sim.masked_fill(self_mask, float("-inf"))
237	
238	    out: dict[int, float] = {}
239	    for k in Ks:
240	        k_eff = min(k, B - 1)
241	        topk_idx = sim_masked.topk(k_eff, dim=1).indices  # (B, k)
242	        retrieved_labels = labels[topk_idx]  # (B, k)
243	        correct = (retrieved_labels == labels.unsqueeze(1)).to(torch.float64)
244	        per_query_precision = correct.mean(dim=1)  # (B,)
245	        # Balance across classes.
246	        per_class_precisions: list[float] = []
247	        for c in range(n_classes):
248	            mask = labels == c
249	            if mask.any():
250	                per_class_precisions.append(per_query_precision[mask].mean().item())
251	        out[k] = (
252	            float(sum(per_class_precisions) / len(per_class_precisions))
253	            if per_class_precisions
254	            else 0.0
255	        )
256	    return out
