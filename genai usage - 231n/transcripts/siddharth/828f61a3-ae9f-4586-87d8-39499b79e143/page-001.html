115	## CNN probe head — design
116	
117	The flat MLP probe has no inductive bias for the structure that THINGS-MEG decoding actually exploits (Cichy / Hebart papers: channel-topographic patterns over time). A 2-layer MLP on flattened tokens may simply lack the right shape; per Dixen 2024 even on EEG, EEGNet/Conformer-style temporal convolutions massively outperform flat heads at this SNR.
118	
119	The new `cnn` head exploits the token tensor's natural structure rather than flattening it:
120	
121	- **For BrainOmni tokens** of shape `(B, 16 latents, 8 time, D=512 embed)`: a small **2-D CNN** with conv over `(latent × time)` and `D` as channels. Two conv blocks with batch-norm + ReLU + global average pooling → linear classifier. ~150k params.
122	- **For raw MEG** of shape `(B, 271 channels, 281 time)`: a small **1-D CNN** over time, channels as input dims. Two strided conv blocks → global average pooling → linear. ~50k params.
123	- **For random tokens**: same shape transform as `tokens` (random codes → `tokens_to_embedding` → CNN). Tests "what does the CNN do with uninformative tokens of the right shape?".
124	- **For μ-transform-style position-preserving tokens** `(B, 1, channels × time)`: reshape to `(B, channels, time)` then use the 1-D CNN — equivalent to applying the raw-MEG CNN to the dequantized signal.
125	
126	Rationale: small enough to not overfit on ~4k training trials per fold, while giving the probe the same inductive bias (temporal smoothness + latent-position locality) that real MEG decoders use.

192	The cheap diagnostic from `meg/CLAUDE.md` §9: take the **existing 3b_nonavg** checkpoint and feed it **image-averaged trials at eval time**. 195	
196	> **A note on reconstruction MSE under averaging.** Time-domain MSE drops 1.07 → 0.29, but this is **not** a 3.6× improvement in the tokenizer's reconstruction quality — it's mostly the **target's noise floor falling**. BrainOmni's preprocess per-trial-per-channel z-scores both single and averaged inputs to unit variance, but the *composition* of that variance differs: in a single z-scored trial ~90% of variance is per-trial noise; in an N-rep average that fraction shrinks by 1/N before z-scoring. The bottleneck drops noise either way; with less noise in the target there's less for the tokenizer to "miss." The honest cross-regime metric is **per-channel Pearson r**, which barely moves: 0.870 (Exp 1) → 0.859 (Exp 1.5). So the right reading is "averaging gives the *probe* a cleaner target," not "the tokenizer reconstructs better" — and the probe results below confirm that the *tokens* see none of that benefit. (General rule: MSE is comparable across checkpoints under fixed input regime, NOT across input regimes or across tokenizer families.)

def _npy_bytes(arr) -> bytes:
    import numpy as np

    buf = io.BytesIO()
    np.save(buf, np.ascontiguousarray(arr), allow_pickle=False)
    return buf.getvalue()


def _write_image_shard(out_path: str, entries: list[tuple[str, bytes]]) -> None:
    tmp = out_path + ".tmp"
    os.makedirs(os.path.dirname(out_path), exist_ok=True)
    with tarfile.open(tmp, "w") as tar:
        for image_id, data in sorted(entries, key=lambda x: x[0]):
            info = tarfile.TarInfo(name=f"{image_id}.npy")
            info.size = len(data)
            tar.addfile(info, io.BytesIO(data))
    os.replace(tmp, out_path)

def _load_token_cache() -> tuple[dict, "np.ndarray", "np.ndarray"]:
    import numpy as np

    cfg = json.loads(open(os.path.join(TOKENS_DIR, "config.json")).read())
    if cfg.get("averaging") != "cross_subject":
        raise RuntimeError(
            f"tokens config does not declare averaging=cross_subject: {cfg.get('averaging')!r}"
        )

    tokens = z["tokens"]

    if tokens.dtype != np.int16 or tokens.shape[1:] != TRIAL_SHAPE:

        )
    if image_ids.shape[0] != tokens.shape[0]:
        raise RuntimeError("tokens / image_ids length mismatch")
    return cfg, tokens, image_ids

def pack_remote() -> dict:
    import numpy as np

    if not os.path.exists(PLAN_PATH):
        raise FileNotFoundError(f"Run plan first; missing {PLAN_PATH}")

    train_set, val_set = _load_split()
    cfg, tokens, image_ids = _load_token_cache()
    token_by_image: dict[str, "np.ndarray"] = {
        str(image_ids[i]): tokens[i] for i in range(tokens.shape[0])
    }

    train_pack = things_manifest.pack_by_catalog_slot(sorted(train_set))
    val_pack = things_manifest.pack_by_catalog_slot(sorted(val_set))

    shutil.rmtree(STAGING_ROOT, ignore_errors=True)
    sentinel = np.full(SENTINEL_SHAPE, -1, dtype=np.int16)
    mask_present = np.array([1], dtype=np.uint8)
    mask_absent = np.array([0], dtype=np.uint8)

    per_split_summary = {}
    for split_name, id_pack in (("train", train_pack), ("val", val_pack)):
        n_real = n_sentinel = 0
        shard_summary: dict[str, dict] = {}
        for slot_id, slot_image_ids in sorted(id_pack.items()):
            tok_entries: list[tuple[str, bytes]] = []
            mask_entries: list[tuple[str, bytes]] = []
            shard_real = 0
            for image_id in slot_image_ids:
                if image_id in token_by_image:
                    block = token_by_image[image_id].reshape(SENTINEL_SHAPE).astype(np.int16, copy=False)
                    tok_entries.append((image_id, _npy_bytes(block)))
                    mask_entries.append((image_id, _npy_bytes(mask_present)))
                    shard_real += 1
                else:
                    tok_entries.append((image_id, _npy_bytes(sentinel)))
                    mask_entries.append((image_id, _npy_bytes(mask_absent)))
            n_real += shard_real
            n_sentinel += len(slot_image_ids) - shard_real
            shard_summary[slot_id] = {
                "n_entries": len(slot_image_ids),
                "n_real": shard_real,
            }

            _write_image_shard(
                os.path.join(_stage_dst(split_name, TOK_MOD), f"{slot_id}.tar"),
                tok_entries,
            )
            _write_image_shard(
                os.path.join(_stage_dst(split_name, MASK_MOD), f"{slot_id}.tar"),
                mask_entries,
            )
        per_split_summary[split_name] = {
            "n_images": sum(len(v) for v in id_pack.values()),
            "n_real": n_real,
            "n_sentinel": n_sentinel,
            "n_shards": len(id_pack),
            "shards": shard_summary,
        }
        print(
            f"[pack_avg] {split_name}: {len(id_pack)} shards, "
            f"{n_real} real / {n_sentinel} sentinel"
        )

    project_volume.commit()

    # Swap staged -> final
    for split_name in ("train", "val"):
        for mod in (TOK_MOD, MASK_MOD):
            staged = _stage_dst(split_name, mod)
            final = _dst(split_name, mod)
            os.makedirs(final, exist_ok=True)
            expected = set(os.listdir(staged))
            for fname in os.listdir(final):
                if fname.endswith(".tar") and fname not in expected:
                    os.remove(os.path.join(final, fname))
            for fname in sorted(os.listdir(staged)):
                os.replace(os.path.join(staged, fname), os.path.join(final, fname))

    shutil.rmtree(STAGING_ROOT, ignore_errors=True)

    # Manifests
    for split_name, manifest_path in (("train", TRAIN_MANIFEST), ("val", VAL_MANIFEST)):
        info = per_split_summary[split_name]
        manifest = {
            "split": split_name,
            "split_source": THINGS_SPLIT_PATH,
            "n_shards": info["n_shards"],
            "n_images": info["n_images"],
            "n_real": info["n_real"],
            "n_sentinel": info["n_sentinel"],
            "shards_subpath": f"things/{TOK_MOD}",
            "mask_subpath": f"things/{MASK_MOD}",
            "tokenizer_cfg": cfg,
            "shards": info["shards"],
        }
        os.makedirs(os.path.dirname(manifest_path), exist_ok=True)
        open(manifest_path, "w").write(json.dumps(manifest, indent=2))

    project_volume.commit()

    return per_split_summary

def verify_remote() -> dict:
    import numpy as np

    train_set, val_set = _load_split()
    train_manifest = json.loads(open(TRAIN_MANIFEST).read())
    val_manifest = json.loads(open(VAL_MANIFEST).read())

    for split_name, manifest, expected_ids in (
        ("train", train_manifest, train_set),
        ("val", val_manifest, val_set),
    ):
        tok_dir = _dst(split_name, TOK_MOD)
        mask_dir = _dst(split_name, MASK_MOD)
        tok_files = sorted(f for f in os.listdir(tok_dir) if f.endswith(".tar"))
        mask_files = sorted(f for f in os.listdir(mask_dir) if f.endswith(".tar"))
        expected_files = sorted(f"{sid}.tar" for sid in manifest["shards"])
        if tok_files != expected_files or mask_files != expected_files:
            raise RuntimeError(f"{split_name}: shard list mismatch")

        seen_ids: set[str] = set()
        n_real = n_sentinel = 0
        for shard_id in manifest["shards"]:
            tok_path = os.path.join(tok_dir, f"{shard_id}.tar")
            mask_path = os.path.join(mask_dir, f"{shard_id}.tar")

            with tarfile.open(tok_path, "r") as tar:
                tok_members = {m.name: tar.extractfile(m).read() for m in tar if m.name.endswith(".npy")}
            with tarfile.open(mask_path, "r") as tar:
                mask_members = {m.name: tar.extractfile(m).read() for m in tar if m.name.endswith(".npy")}

            if set(tok_members) != set(mask_members):
                raise RuntimeError(f"{split_name}/{shard_id}: tok / mask key mismatch")

            for fname, raw in tok_members.items():
                image_id = fname[:-4]
                if image_id not in expected_ids:
                    raise RuntimeError(f"{split_name}/{shard_id}: {image_id} not in split")
                seen_ids.add(image_id)
                arr = np.load(io.BytesIO(raw), allow_pickle=False)
                if arr.shape != SENTINEL_SHAPE or arr.dtype != np.int16:
                    raise RuntimeError(f"{fname}: bad token array {arr.shape} {arr.dtype}")
                marr = np.load(io.BytesIO(mask_members[fname]), allow_pickle=False)
                if marr.shape != (1,) or marr.dtype != np.uint8 or marr[0] not in (0, 1):
                    raise RuntimeError(f"{fname}: bad mask array {marr.shape} {marr.dtype} {marr}")
                is_sentinel = bool((arr == -1).all())
                expected_present = int(marr[0] == 1)
                if is_sentinel == bool(expected_present):
                    raise RuntimeError(
                        f"{fname}: token sentinel ({is_sentinel}) inconsistent with mask ({marr[0]})"
                    )
                if is_sentinel:
                    n_sentinel += 1
                else:
                    n_real += 1

        if seen_ids != expected_ids:
            raise RuntimeError(f"{split_name}: image_ids != things_split set")
        if n_real != manifest["n_real"] or n_sentinel != manifest["n_sentinel"]:
            raise RuntimeError(
                f"{split_name}: real/sentinel counts mismatch manifest"
            )
        print(
            f"[verify_avg] {split_name}: {len(tok_files)} shards, "
            f"{n_real} real / {n_sentinel} sentinel — OK"
        )

    return {
        "train_n_shards": train_manifest["n_shards"],
        "val_n_shards": val_manifest["n_shards"],
        "train_n_real": train_manifest["n_real"],
        "val_n_real": val_manifest["n_real"],
    }
