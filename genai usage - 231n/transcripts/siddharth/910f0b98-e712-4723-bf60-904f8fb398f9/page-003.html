def _train_and_score_probe(
    feats: torch.Tensor,
    labels: torch.Tensor,
    train_idx: torch.Tensor,
    test_idx: torch.Tensor,
    n_classes: int,
    config: EvalConfig,
) -> dict[str, float]:
    """Fit a single nn.Linear with full-batch gradient descent.

    Returns a dict of named metrics: `top{k}` for each k in probe_top_k, and
    `bal_acc` (balanced accuracy = mean of per-class recalls). Balanced acc
    is the imbalance-invariant diagnostic: chance is exactly 1/n_classes
    regardless of class frequencies, whereas top-1 floor is `max(class_freq)`
    once a classifier collapses to majority prediction.
    """
    device = config.device
    X_train = feats[train_idx].to(device)
    y_train = labels[train_idx].to(device)
    X_test = feats[test_idx].to(device)
    y_test = labels[test_idx].to(device)

    feat_dim = feats.shape[1]
    torch.manual_seed(config.seed)
    head = torch.nn.Linear(feat_dim, n_classes).to(device)
    opt = torch.optim.AdamW(

    )

def _topk_accuracy(
    logits: torch.Tensor, labels: torch.Tensor, ks: tuple[int, ...]
) -> dict[int, float]:
    """Top-k accuracy. Falls back to top-min(k, n_classes) silently."""

        correct = (top == labels.unsqueeze(1)).any(dim=1).to(torch.float64)
        out[k] = float(correct.mean().item())
    return out


def _balanced_accuracy(
    logits: torch.Tensor, labels: torch.Tensor, n_classes: int
) -> float:
    """Mean of per-class recalls. Invariant to class imbalance.

    Classes absent from the test set contribute nothing (averaged over
    only the classes that *do* appear). This matters when a fold happens
    to miss a tiny class — the alternative (count 0 recall for missing
    classes) penalizes folds for sampling variance, not classifier quality.
    """
    preds = logits.argmax(dim=1)
    per_class_recall: list[float] = []
    for c in range(n_classes):
        mask = labels == c
        if mask.any():
            per_class_recall.append((preds[mask] == c).to(torch.float64).mean().item())
    return float(sum(per_class_recall) / len(per_class_recall)) if per_class_recall else 0.0

Your lean toward C is correct. Here's the analysis with our specific data in mind.

C dominates B on every axis

Axis B (subsample) C (class weights)
Data efficiency Subsamples to smallest class. With drink ≈ 1.6% of trials and 27 classes, we'd cap at ~60 train trials/class → ~1620 total per fold. ~60% data loss. Uses all 4,000 training trials per fold. Zero data loss.
Variance High — result depends on which random subset got drawn Low — every sample contributes
Conceptual effect on gradients Each class gets equal sample count in the loss Each class gets equal gradient mass in the loss (mathematically equivalent in expectation)
Compute Same forward/backward, fewer batches Same forward/backward, all batches
Code change Subsample indices per fold before training One weight= kwarg to F.cross_entropy
Standard ML practice Rarely used outside very imbalanced settings sklearn / PyTorch default for imbalance

The key statistical fact: C and B optimize the same objective in expectation, but C does it with the lower-variance estimator. Subsampling literally throws away information that class weights preserve.

The only scenarios B beats C: - Framework doesn't support sample weights (PyTorch obviously does) - You want exact-symmetric class counts for some downstream test (we don't) - Class-conditional feature distributions are so different that reweighting is dangerous (not our case — same features, just different label frequencies)

None of these apply.

One implementation detail worth flagging

Class weights should be computed per fold, from the training labels of that fold — not globally from all labels. sklearn's "balanced" formula:

# n_classes = 27, train_labels of shape (4000,)
counts = torch.bincount(train_labels, minlength=n_classes).float()
weights = train_labels.shape[0] / (n_classes * counts.clamp(min=1))
# any class missing from this fold gets weight = N_train / N_classes (defensive)
loss = F.cross_entropy(logits, train_labels, weight=weights.to(device))

This is the most common formula and is what sklearn applies under class_weight='balanced'.

def _class_weights(
    y_train: torch.Tensor, n_classes: int, device: torch.device | str
) -> torch.Tensor:
    """sklearn-style balanced weights: w_c = N_total / (n_classes * count_c).

    Computed per-fold from the training labels (not globally), so each fold's
    weights match its own class distribution. Classes absent from this fold
    fall back to the unweighted-uniform default (N_total / n_classes) — a
    defensive choice that prevents `0 * +inf` if any class happens to be
    missing on a small fold.
    """
    counts = torch.bincount(y_train, minlength=n_classes).to(torch.float64)
    n_total = float(y_train.shape[0])
    default = n_total / n_classes
    weights = torch.where(
        counts > 0,
        n_total / (n_classes * counts.clamp(min=1.0)),
        torch.full_like(counts, default),
    )
    return weights.to(device=device, dtype=torch.float32)
